4.Write a program to demonstrate the do while loop
package pack1;

public class DOwhilelop {

	public static void main(String[] args) {
		int i= 0;
		System.out.println("do while loop");
		do
		{
			System.out.println("iteration i=" +i);
			i++;
		}while(i<=9);

	}

}
