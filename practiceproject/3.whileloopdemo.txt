3.Write a program to demonstrate the while loop

package pack1;

public class Whileprogram {

	public static void main(String[] args) {
		int i=1;
		System.out.println("whileloop program");
		while(i<=7)
		{
			
			System.out.println("iteration is =" +i);
			i++;
		}

	}

}
