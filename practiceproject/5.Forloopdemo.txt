5.Write a program to demonstrate the for loop
package pack1;

public class Forloop {

	public static void main(String[] args) {
		System.out.println("for loop demonstration");
	
		for( int i=0;i<=11;i++)
		{
			System.out.println("Iteration i = " +i);
		}

	}

}
